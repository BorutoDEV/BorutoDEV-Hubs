-- === ANTI-LOG SYSTEM ===
local AntiLogEnabled = true

-- Block common logging remotes
local function blockLoggingRemote(remote, name)
    if remote then
        local oldFireServer = remote.FireServer
        remote.FireServer = function(self, ...)
            if AntiLogEnabled then
                local args = {...}
                -- Block analytics/log remotes (common patterns)
                if name:lower():find("log") or name:lower():find("analytics") or 
                   name:lower():find("report") or name:lower():find("telemetry") or
                   (typeof(args[1]) == "string" and (args[1]:lower():find("exploit") or args[1]:lower():find("cheat"))) then
                    print("[Anti-Log] Blocked: " .. name)
                    return nil
                end
            end
            return oldFireServer(self, ...)
        end
        
        local oldInvokeServer = remote.InvokeServer
        remote.InvokeServer = function(self, ...)
            if AntiLogEnabled then
                local args = {...}
                if name:lower():find("log") or name:lower():find("analytics") or 
                   name:lower():find("report") or name:lower():find("check") then
                    print("[Anti-Log] Blocked invoke: " .. name)
                    return nil
                end
            end
            return oldInvokeServer(self, ...)
        end
    end
end

-- Hook all remotes in ReplicatedStorage
task.spawn(function()
    task.wait(1)
    local RS = game:GetService("ReplicatedStorage")
    for _, obj in ipairs(RS:GetDescendants()) do
        if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
            blockLoggingRemote(obj, obj.Name)
        end
    end
    
    -- Hook future remotes
    RS.DescendantAdded:Connect(function(obj)
        if obj:IsA("RemoteEvent") or obj:IsA("RemoteFunction") then
            task.wait(0.1)
            blockLoggingRemote(obj, obj.Name)
        end
    end)
end)

-- Clear client logs
local function clearLogs()
    local LogService = game:GetService("LogService")
    if LogService.ClearOutput then
        pcall(function() LogService:ClearOutput() end)
    end
end

-- Auto clear logs periodically
task.spawn(function()
    while true do
        if AntiLogEnabled then
            clearLogs()
        end
        task.wait(30)
    end
end)

-- Spoof client info
local function spoofInfo()
    local mt = getrawmetatable(game)
    if mt then
        local oldNamecall = mt.__namecall
        setreadonly(mt, false)
        
        mt.__namecall = newcclosure(function(self, ...)
            local method = getnamecallmethod()
            if AntiLogEnabled and (method == "Kick" or method == "kick") then
                print("[Anti-Log] Blocked kick attempt")
                return nil
            end
            return oldNamecall(self, ...)
        end)
        
        setreadonly(mt, true)
    end
end

pcall(spoofInfo)

local function stringToCFrame(str)
    if not str then return CFrame.new(0,0,0) end
    local x, y, z = str:match("([^,]+),%s*([^,]+),%s*([^,]+)")
    if x and y and z then
        return CFrame.new(tonumber(x) or 0, tonumber(y) or 0, tonumber(z) or 0)
    end
    return CFrame.new(0,0,0)
end

local function teleportToTarget(cf, duration)
    local hrp = getRootPart()
    if not hrp then return end
    if duration and duration > 0 then
        local ts = game:GetService("TweenService")
        local info = TweenInfo.new(duration, Enum.EasingStyle.Linear)
        local goal = { CFrame = cf }
        ts:Create(hrp, info, goal):Play()
    else
        hrp.CFrame = cf
    end
end

local killAuraToggle = false
local radius = 200

local toolsDamageIDs = {
    ["Old Axe"] = "1_8982038982",
    ["Good Axe"] = "112_8982038982",
    ["Strong Axe"] = "116_8982038982",
    ["Chainsaw"] = "647_8992824875",

local function getAnyToolWithDamageID()
    local inventory = LocalPlayer:FindFirstChild("Inventory")
    if not inventory then return nil, nil end
    for toolName, damageID in pairs(toolsDamageIDs) do
        local tool = inventory:FindFirstChild(toolName)
        if tool then return tool, damageID end
    end
    return nil, nil
end

local function equipTool(tool)
    if tool and remoteEvents then
        remoteEvents.EquipItemHandle:FireServer("FireAllClients", tool)
    end
end

local function unequipTool(tool)
    if tool and remoteEvents then
        remoteEvents.UnequipItemHandle:FireServer("FireAllClients", tool)
    end
end

local function killAuraLoop()
    while killAuraToggle do
        local hrp = getRootPart()
        if hrp and characterFolder and remoteEvents then
            local tool, damageID = getAnyToolWithDamageID()
            if tool and damageID then
                equipTool(tool)
                for _, mob in ipairs(characterFolder:GetChildren()) do
                    if mob:IsA("Model") then
                        local part = mob:FindFirstChildWhichIsA("BasePart")
                        if part and (part.Position - hrp.Position).Magnitude <= radius then
                            pcall(function()
                                remoteEvents.ToolDamageObject:InvokeServer(mob, tool, damageID, CFrame.new(part.Position))
                            end)
                        end
                    end
                end
                task.wait(0.1)
            else
                task.wait(1)
            end
        else
            task.wait(0.5)
        end
    end
end

MainSection:NewToggle("Kill Aura", "Auto attack nearby mobs", function(state)
    killAuraToggle = state
    if state then task.spawn(killAuraLoop) else
        local tool, _ = getAnyToolWithDamageID()
        unequipTool(tool)
    end
end)

    if not item then return nil end
    for _, descendant in ipairs(item:GetDescendants()) do

        if descendant:IsA("Model") then
            for _, sub in ipairs(descendant:GetDescendants()) do

            end

    return nil
end

-- Auto Bring Trees
local originalTreeCFrames = {}
local treesBrought = false

local function getAllSmallTrees()
    local trees = {}
    local map = workspace:FindFirstChild("Map")
    if not map then return trees end
    local foliage = map:FindFirstChild("Foliage")
    local landmarks = map:FindFirstChild("Landmarks")
    
    local function scan(folder)
        if not folder then return end
        for _, obj in ipairs(folder:GetChildren()) do
            if obj:IsA("Model") and obj.Name == "Small Tree" then
                table.insert(trees, obj)
            end
        end
    end
    scan(foliage)
    scan(landmarks)
    return trees
end

local function findTrunk(tree)
    if not tree then return nil end
    for _, part in ipairs(tree:GetDescendants()) do
        if part:IsA("BasePart") and part.Name == "Trunk" then return part end
    end
    return nil
end

TreeSection:NewToggle("Auto Bring All Small Trees", "Bring trees to you", function(state)
    if state and not treesBrought then
        local hrp = getRootPart()
        if not hrp then return end
        local target = CFrame.new(hrp.Position + hrp.CFrame.LookVector * 10)
        for _, tree in ipairs(getAllSmallTrees()) do
            local trunk = findTrunk(tree)
            if trunk then
                if not originalTreeCFrames[tree] then originalTreeCFrames[tree] = trunk.CFrame end
                tree.PrimaryPart = trunk
                trunk.Anchored = false
                trunk.CanCollide = false
                task.wait()
                pcall(function() tree:SetPrimaryPartCFrame(target + Vector3.new(math.random(-5,5), 0, math.random(-5,5))) end)
                trunk.Anchored = true
            end
        end
        treesBrought = true
    elseif not state and treesBrought then
        for tree, cframe in pairs(originalTreeCFrames) do
            if tree and tree.Parent then
                local trunk = findTrunk(tree)
                if trunk then
                    tree.PrimaryPart = trunk
                    pcall(function() tree:SetPrimaryPartCFrame(cframe) end)
                    trunk.Anchored = true
                    trunk.CanCollide = true
                end
            end
        end
        originalTreeCFrames = {}
        treesBrought = false
    end
end)
